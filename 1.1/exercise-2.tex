$Let \quad m, n \in  N, \quad m < n$\\
$
Having \quad m = kn + r \quad and \quad m < n\\
\Rightarrow (k = 0 \quad and \quad r =  m)\\
Having \quad m \leftarrow n, n \leftarrow r \quad and \quad m < n\\
\Rightarrow m > n\\
$
